253 static void stepHandlerFn(clkgen_t clk, void *data_)
254 {
255 struct StepperMotorTimerData *data = (struct StepperMotorTimerData *)data_;
256 StepperMotor *_this = data->_this;
257 if (data->steps && !(*(_this->done)))
258 {
259 _this->onestep(data->dir, data->style);
260 data->steps--;
261 }
262 if (data->steps == 0 || (*(_this->done)))
263 {
264 _this->cond.notify_all();
265 }
266 }
